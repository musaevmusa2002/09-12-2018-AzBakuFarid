Regionlar üzrə işçilərin sayını göstərin (region, employee)

select custom.regiondescription, count(custom.EmployeeID) as employees from (select distinct et.EmployeeID, r.regiondescription from region r join Territories t on r.RegionID=t.RegionID join EmployeeTerritories et on t.TerritoryID=et.TerritoryID) custom group by custom.regiondescription

Məhsulları ən çox olan ilk 3 kateqoriyanı göstərin (categories, products)

select top 3 categoryname, summ from Categories c join  (select sum(UnitsInStock) as summ, CategoryID  from Products group by CategoryID ) custom on c.CategoryID=custom.CategoryID order by summ desc
